New Terminal

Sunday, 11 November 2007

tell application "System Events"
    if (count (processes whose name is "Terminal")) is 0 then
        tell application "Terminal"
            activate
        end tell
    else
        tell application "Terminal"
            activate
            do script ""
        end tell
    end if
end tell
